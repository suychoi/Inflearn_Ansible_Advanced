기초, 한 번의 명령으로 다수의 시스템에 작업하기

# shell 모듈 사용
ansible all -m shell -a "uptime" -k
ansible all -m shell -a "df -h"
ansible all -m shell -a "free -h" -k

# 사용자 생성 모듈 생성후 암호,복호화를 진행해야 사용 가능
ansible all -m user -a "name=suychoi password=1234" -k

# copy 모듈
ansible nginx -m copy -a "src=./bloter.file dest=/tmp/" -k
ansible nginx -m copy -a "src=/etc/resolv.conf dest=/etc/resolv.conf" -k

# Centos yum module 사용
ansible nginx -m yum -a "name=httpd status=present" -k -K

SuperPutty

-K : 관리자 권한으로 실행
